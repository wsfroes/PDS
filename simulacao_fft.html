freq_ruido = 60

# Adicionando uma senoide + um componente aleatório (ruído branco)
ruido_base = 1.2 * np.sin(2 * np.pi * freq_ruido * t)
ruido_aleatorio = 0.2 * np.random.normal(0, 1, len(t)) # O '0.2' é o volume da sujeira
ruido_sujo = ruido_base + ruido_aleatorio

# Aplicando a janela de Hamming para reduzir o vazamento espectral
janela = hamming(N)
sinal_janelado = sinal_capturado * janela

# O Software calcula a Transformada Rápida (Adeus contas na mão!)
# FFT e Normalização Correta

valores_fft = fft(sinal_janelado)
frequencias = fftfreq(N, 1/Fs)

# Normalização de Energia (Amp = 2 * abs(fft) / sum(janela))
# Isso garante que a amplitude medida corresponda à amplitude real da senoide
amplitude_fft = 2 * np.abs(valores_fft[:N//2]) / np.sum(janela)
